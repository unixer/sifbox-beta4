#!/bin/sh

. config/options

cd $BUILD/$1*
ln -s $SYSROOT_PREFIX/usr/local/lib/libusb-1.0.so.0.1.0 $SYSROOT_PREFIX/usr/local/lib/libusb-1.0.a
mkdir -p build && cd build

cmake -DCMAKE_TOOLCHAIN_FILE=$CMAKE_CONF \
      -DCMAKE_INSTALL_PREFIX=/usr \
      -DLIBUSBDIR=$SYSROOT_PREFIX/usr/local \
      -DLIBUSB=$SYSROOT_PREFIX/usr/local \
      -DWITH_SSL=1 -DWITH_SSLv3=1 \
      -DHAVE_DVBAPI=1 -DWITH_STAPI=0 \
      -DMODULE_CAMD33=1 \
      -DWEBIF=1 \
      -DOPTIONAL_INCLUDE_DIR=$SYSROOT_PREFIX/usr/local/include \
      ..

make
make_install
